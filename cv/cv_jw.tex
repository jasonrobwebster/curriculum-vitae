\newcommand*{\skillbar}[1]{%
    \skillb@level=\dimexpr#1\skillb@length/100\relax%

    {\color{noskillgray}%
        \rule{\dimexpr\skillb@length-\skillb@level\relax}{\skillb@height}}%
}
